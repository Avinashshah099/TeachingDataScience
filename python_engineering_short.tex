import numpy
import matplotlib.pyplot as plt

x = numpy.linspace(-15,15,100) # 100 linearly spaced numbers
y = numpy.sin(x)/x # computing the values of sin(x)/x

plt.plot(x,y) # sin(x)/x
plt.show() # show the plot
